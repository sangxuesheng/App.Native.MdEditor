# 微信公众号格式导出 - 优化方案



## 核心模块

### 1. 模板管理模块（Template Store）
- 管理 Markdown 模板
- 导出模板为 JSON
- 支持模板的增删改查

### 2. 主题导出模块（Theme Exporter）
- 导出微信专属 CSS 样式
- 合并基础主题 + 自定义样式
- 支持 CSS 变量配置

### 3. 内容转换模块（Content Converter）
- Markdown 转 HTML
- 应用微信样式
- 处理图片、代码块等特殊元素

```javascript
// 主题配置
const wechatThemes = {
  default: {
    primary: '#3daeff',
    secondary: '#f7f7f7',
    text: '#333',
    code: '#f6f8fa'
  },
  green: {
    primary: '#52c41a',
    secondary: '#f6ffed',
    text: '#333',
    code: '#f6ffed'
  },
  orange: {
    primary: '#fa8c16',
    secondary: '#fff7e6',
    text: '#333',
    code: '#fff7e6'
  }
}

// 应用主题
const convertToWechatFormat = (html, themeName = 'default') => {
  const theme = wechatThemes[themeName]
  const wechatStyles = generateStyles(theme)
  return `${wechatStyles}<div class="wechat-content">${html}</div>`
}

// 生成样式
const generateStyles = (theme) => {
  return `
    <style>
      .wechat-content {
        --primary-color: ${theme.primary};
        --secondary-color: ${theme.secondary};
        --text-color: ${theme.text};
        --code-bg: ${theme.code};
      }
      
      .wechat-content h1 {
        border-bottom-color: var(--primary-color);
      }
      
      .wechat-content h2 {
        border-left-color: var(--primary-color);
      }
      
      .wechat-content blockquote {
        background: var(--secondary-color);
        border-left-color: var(--primary-color);
      }
      
      .wechat-content pre {
        background: var(--code-bg);
      }
    </style>
  `
}
```

```typescript
interface WechatTemplate {
  id: string
  name: string
  content: string  // Markdown 内容
  theme: string    // 主题名称
  description?: string
  tags?: string[]
  createdAt: number
  updatedAt: number
}

// 模板管理
class TemplateManager {
  templates: WechatTemplate[] = []
  
  // 添加模板
  addTemplate(template: WechatTemplate) {
    this.templates.push(template)
    this.save()
  }
  
  // 导出模板
  exportTemplates(): string {
    return JSON.stringify(this.templates, null, 2)
  }
  
  // 导入模板
  importTemplates(json: string) {
    this.templates = JSON.parse(json)
    this.save()
  }
  
  // 保存到 localStorage
  save() {
    localStorage.setItem('wechat-templates', JSON.stringify(this.templates))
  }
}
```

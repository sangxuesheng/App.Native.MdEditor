## 解决方案

### 方案 1：添加全局快捷键监听

在 App.jsx 中添加全局快捷键处理：

```javascript
// 在 App 组件中添加
useEffect(() => {
  const handleGlobalKeyDown = (e) => {
    // Ctrl+N - 新建文件
    if (e.ctrlKey && e.key === 'n' && !e.shiftKey) {
      e.preventDefault()
      handleNewFile()
    }
    
    // Ctrl+Shift+S - 另存为
    if (e.ctrlKey && e.shiftKey && e.key === 'S') {
      e.preventDefault()
      handleSaveAs()
    }
    
    // Ctrl+T - 切换主题
    if (e.ctrlKey && e.key === 't') {
      e.preventDefault()
      toggleEditorTheme()
    }
    
    // Shift+Alt+F - 格式化文档
    if (e.shiftKey && e.altKey && e.key === 'F') {
      e.preventDefault()
      handleMenuFormatDocument()
    }
    
    // Ctrl+1-6 - 插入标题
    if (e.ctrlKey && /^[1-6]$/.test(e.key)) {
      e.preventDefault()
      const level = parseInt(e.key)
      handleToolbarInsert('#'.repeat(level) + ' ', '', 'heading')
    }
  }
  
  window.addEventListener('keydown', handleGlobalKeyDown)
  return () => window.removeEventListener('keydown', handleGlobalKeyDown)
}, [handleNewFile, handleSaveAs, toggleEditorTheme, handleMenuFormatDocument, handleToolbarInsert])
```

### 方案 3：快捷键冲突解决

**问题**: Ctrl+B 同时用于"加粗"和"切换文件树"

**解决方案**:
1. Ctrl+B - 保留为加粗（编辑器内）
2. Ctrl+\\ - 改为切换文件树（VS Code 风格）
3. 或 F2 - 切换文件树

## 实现代码

### 完整的全局快捷键处理

```javascript
// 在 App.jsx 中添加
useEffect(() => {
  const handleGlobalKeyDown = (e) => {
    // 如果焦点在输入框中，某些快捷键不处理
    const isInputFocused = document.activeElement.tagName === 'INPUT' || 
                          document.activeElement.tagName === 'TEXTAREA'
    
    // Ctrl+N - 新建文件
    if (e.ctrlKey && e.key === 'n' && !e.shiftKey && !isInputFocused) {
      e.preventDefault()
      handleNewFile()
      return
    }
    
    // Ctrl+Shift+S - 另存为
    if (e.ctrlKey && e.shiftKey && e.key === 'S') {
      e.preventDefault()
      handleSaveAs()
      return
    }
    
    // Ctrl+T - 切换主题
    if (e.ctrlKey && e.key === 't' && !isInputFocused) {
      e.preventDefault()
      toggleEditorTheme()
      return
    }
    
    // Ctrl+\\ - 切换文件树
    if (e.ctrlKey && e.key === '\\') {
      e.preventDefault()
      setShowFileTree(prev => !prev)
      return
    }
    
    // Shift+Alt+F - 格式化文档
    if (e.shiftKey && e.altKey && e.key === 'F') {
      e.preventDefault()
      if (editorRef.current) {
        handleMenuFormatDocument()
      }
      return
    }
    
    // Ctrl+1-6 - 插入标题（仅在编辑器有焦点时）
    if (e.ctrlKey && /^[1-6]$/.test(e.key) && editorRef.current) {
      e.preventDefault()
      const level = parseInt(e.key)
      handleToolbarInsert('#'.repeat(level) + ' ', '', 'heading')
      return
    }
  }
  
  window.addEventListener('keydown', handleGlobalKeyDown)
  return () => window.removeEventListener('keydown', handleGlobalKeyDown)
}, [handleNewFile, handleSaveAs, toggleEditorTheme, handleMenuFormatDocument, handleToolbarInsert, setShowFileTree])
```

### 阶段 3：useMemo 优化计算密集型操作

#### 需要优化的计算

```javascript
// 1. Markdown 处理
const markdownHtml = useMemo(() => {
  if (!content) return ''
  try {
    return processor.processSync(content).toString()
  } catch (error) {
    console.error('Markdown processing error:', error)
    return '<p>Markdown 处理错误</p>'
  }
}, [content]) // 只在 content 变化时重新计算

// 2. 文件树数据
const fileTreeData = useMemo(() => {
  return buildFileTree(rootDirs)
}, [rootDirs])

// 3. 大纲数据
const outlineData = useMemo(() => {
  return extractHeadings(content)
}, [content])

// 4. 搜索结果
const searchResults = useMemo(() => {
  if (!searchQuery) return []
  return filterFiles(files, searchQuery)
}, [files, searchQuery])
```

2. 用 `useMemo` 包裹

4. 使用 React DevTools Profiler 验证效果

### 阶段 4：优化 Markdown 处理性能

#### 4.1 防抖优化

```javascript
// 使用 useDebounce 延迟 Markdown 处理
const debouncedContent = useDebounce(content, 300) // 300ms 防抖

const markdownHtml = useMemo(() => {
  return processor.processSync(debouncedContent).toString()
}, [debouncedContent])
```

#### 4.2 条件加载插件

```javascript
// 只在需要时加载数学公式插件
const processor = useMemo(() => {
  const hasMath = content.includes('$') || content.includes('\\(')
  
  return unified()
    .use(remarkParse)
    .use(remarkGfm)
    .use(hasMath ? remarkMath : null) // 条件加载
    .use(remarkRehype)
    .use(hasMath ? rehypeKatex : null) // 条件加载
    .use(rehypeStringify)
}, [content])
```

#### 4.3 Web Worker 处理（高级）

```javascript
// workers/markdown.worker.js
import { unified } from 'unified'
// ... 其他导入

self.onmessage = (e) => {
  const { content } = e.data
  const html = processor.processSync(content).toString()
  self.postMessage({ html })
}

// App.jsx
const worker = useMemo(() => new Worker('/workers/markdown.worker.js'), [])

useEffect(() => {
  worker.postMessage({ content: debouncedContent })
  worker.onmessage = (e) => {
    setMarkdownHtml(e.data.html)
  }
}, [debouncedContent])
```

### 性能监控代码

```javascript
// utils/performance.js
export function measurePerformance(name, fn) {
  const start = performance.now()
  const result = fn()
  const end = performance.now()
  console.log(`[Performance] ${name}: ${(end - start).toFixed(2)}ms`)
  return result
}

// 使用
const html = measurePerformance('Markdown Processing', () => {
  return processor.processSync(content).toString()
})
```

### 3. 检查哪个 CSS 规则在控制 max-width

```javascript
const img = document.querySelector('.markdown-body img[style*="width"]')
if (img) {
  console.log('=== CSS 规则检查 ===')
  
  // 获取所有匹配的规则
  const matchedRules = []
  Array.from(document.styleSheets).forEach(sheet => {
    try {
      Array.from(sheet.cssRules || []).forEach(rule => {
        if (rule.selectorText && img.matches(rule.selectorText)) {
          if (rule.style.maxWidth) {
            matchedRules.push({
              selector: rule.selectorText,
              maxWidth: rule.style.maxWidth,
              important: rule.style.cssText.includes('!important'),
              source: sheet.href || 'inline'
            })
          }
        }
      })
    } catch (e) {}
  })
  
  console.log('影响 max-width 的规则:', matchedRules)
}
```

### 4. 手动测试强制设置

```javascript
const img = document.querySelector('.markdown-body img[style*="width"]')
if (img) {
  console.log('=== 手动强制设置测试 ===')
  console.log('设置前 max-width:', window.getComputedStyle(img).maxWidth)
  
  // 强制设置
  img.style.maxWidth = 'none'
  img.style.display = 'inline-block'
  
  console.log('设置后 max-width:', window.getComputedStyle(img).maxWidth)
  console.log('设置后 display:', window.getComputedStyle(img).display)
  console.log('设置后宽度:', img.offsetWidth + 'px')
  
  console.log('✅ 如果图片大小改变了，说明 CSS 选择器有问题')
  console.log('❌ 如果图片大小没变，说明有其他因素影响')
}
```
